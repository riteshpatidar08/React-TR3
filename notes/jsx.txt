// App.jsx

// ✅ Import external CSS file
import './App.css';

function App() {
  const version = 19; // ✅ JavaScript variable

  return (
    <div>
      {/* 
        ✅ Inline Styling in JSX:
        - Use a JavaScript object
        - CSS properties must be written in camelCase
        - Values are typically strings or variables
      */}
      <h1 
        style={{ color: 'red', backgroundColor: 'blue' }} 
        className="heading" // ✅ Use className instead of class in JSX
      >
        REACT : {version} {/* ✅ Use {} to embed JS expressions inside JSX */}
      </h1>

      {/* 
        ✅ Attributes:
        - JSX uses camelCase for HTML attributes (e.g., maxLength, className)
        - Numeric values (like maxLength) should be passed using curly braces
      */}
      <input type="text" maxLength={10} />
    </div>
  );
}

export default App;
